%-Baumdiagramm-Notation-%

%!forest-Baumdiagramm: Standard
\newcommand{\lingforest}[2]{%
	\noindent\makebox[\textwidth]{
		\begin{forest}
			where n children=0{tier=word}{},
			for tree={l=#1pt}
			#2
	\end{forest}}
}

%!forest-Baumdiagramm: Pittner-Notation
\newcommand{\lingforestpittner}[3]{%
	\noindent\makebox[\textwidth]{
		\begin{forest}
			where n children=0{tier=word}{},
			for tree={l=#1pt}
			[,phantom
			#2
			#3
			]
	\end{forest}}
}
